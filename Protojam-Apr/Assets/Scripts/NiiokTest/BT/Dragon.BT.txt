tree("Root")
	fallback
		tree("Die")
		tree("Grab")
		random(9, 1)
			tree("Move")
			tree("Run")


tree("Die")
	sequence
		IsDead()
		SetMoving(false)

tree("Grab")
	sequence
		IsPlayerNear(4.0)
		SetMoving(false)
		Grab()
		Wait(0.02)

tree("Move")
	sequence
		SetMoving(true)
		Wait(0.05)


tree("Run")
	sequence
		SetMoving(false)
		Wait(0.02)
